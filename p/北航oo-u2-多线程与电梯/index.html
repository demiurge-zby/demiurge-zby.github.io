private boolean cannotArrive(Elevator elevator, PersonRequest request) {
        boolean upperCannot =
            (getFloorNumber(request.getFromFloor()) < elevator.getTransferFloorNumber()) ||
            (getFloorNumber(request.getFromFloor()) == elevator.getTransferFloorNumber() && getFloorNumber(request.getToFloor()) < getFloorNumber(request.getFromFloor()));

        boolean lowerCannot =
            (getFloorNumber(request.getFromFloor()) > elevator.getTransferFloorNumber()) ||
            (getFloorNumber(request.getFromFloor()) == elevator.getTransferFloorNumber() && getFloorNumber(request.getToFloor()) > getFloorNumber(request.getFromFloor()));

        return ((elevator.isUpper() && upperCannot)
                || (!elevator.isUpper() && lowerCannot));
    }
